Subrutina EncontrarEquipo (PTR, codigo)
	p <- PTR
	Hacer
		Si(p.cod = codigo)
			Si( p= PTR )
				p.LLinkeq.Rlinkeq<-null
				PTR<-p.RLinkeq
				p.Rlinkeq<-null
			Sino
				Si (P = final)
					end <- p.prev
				    end.RLinkeq <- null
				    p.LLinkeq <- null 
				    p <- end
				Sino
				    p.LLinkeq.RLinkeq <- p.RLinkeq
        			p.next.LLinkeq <-p.LLinkeq
				    p.LLinkeq <- nulo
				    p.RLinkeq <- nulo
				FinSi
			FinSi
		Sino 
			p<- p.RLinkeq
		FinSi
	Hasta (p = final)
FinSubRutina

		    p<-p.RLinkeq

		q.cod <- codigo 
		Si(p = PTR)

			PTR <- q
		Sino

			p.prevE<-q
